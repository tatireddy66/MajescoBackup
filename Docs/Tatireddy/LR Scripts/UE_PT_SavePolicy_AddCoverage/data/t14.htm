function OnLoad()
{
    $('#RelatedClaims_Container').hide();
    $('#tbw_RelClaims').ready(function () {
        var tbwobj1 = $('#tbw_RelClaims');
        initTableWalker('CLAIMS_EGL_PAGE1', tbwobj1, true, 10);
    });
    $('#tbw_CSpolInfo').ready(function () {
        var tbwobj2 = $('#tbw_CSpolInfo');
        initTableWalker('CLAIMS_EGL_PAGE1', tbwobj2, true, 5);
    });

    initTableWalker('CLAIMS_EGL_PAGE1', $('#Lawsuit_Table'), true, 10);
    
    scroll(0,0);

  $("#supervisor_name1").autocomplete("claims_egl_page1.max", {
       selectFirst:false, delay:10,
       max:20,cacheLength:0,
       extraParams: {
           AJAX_ACTION: "GetSupervisors",
           PreProcess: "false",
           ID:"AD09471A1B1F4A118D56834EED7E7D8950B3CF"
       }
  });

  $("#supervisor_name1").bind('result', function (ev, data, v) {
    var resp = v.split('&nbsp;&nbsp;');
    var supId = $(resp[1]).html();
    $('#supervisor_name1').val(resp[0]);
    $('#SupervisorID').attr('value', supId);
  });
}

function SaveSupervisor(){
    var clmid = '70433';
    var supID = $('#SupervisorID').val();

    Communication.CustomRequest('claims_egl_page1.max?AJAX_ACTION=SaveSupervisor&SUPID=' + supID + '&CLMID=' + clmid, function(AjaxResp){
              if (AjaxResp == 1){
                     jAlert('Supervisor is saved successfully!', 'Success');
              }
        });
}
